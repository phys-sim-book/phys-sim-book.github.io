23template <typename T>
24void MatrixFormatConverter<T, 1>::merge_sort_indices_and_values(
25 const DeviceTripletMatrix<T, 1>& from, DeviceBCOOMatrix<T, 1>& to)
26{
27 using namespace muda;
28
29 auto src_row_indices = from.row_indices();
30 auto src_col_indices = from.col_indices();
31 auto src_values = from.values();
32
33 loose_resize(sort_index, src_row_indices.size());
34 loose_resize(ij_pairs, src_row_indices.size());
35
36 ParallelFor(256)
37 .kernel_name("set ij pairs")
38 .apply(src_row_indices.size(),
39 [row_indices = src_row_indices.cviewer().name("row_indices"),
40 col_indices = src_col_indices.cviewer().name("col_indices"),
41 ij_pairs = ij_pairs.viewer().name("ij_pairs")] __device__(int i) mutable
42 {
43 ij_pairs(i).x = row_indices(i);
44 ij_pairs(i).y = col_indices(i);
45 });
46
47 ParallelFor(256)
48 .kernel_name("iota") //
49 .apply(src_row_indices.size(),
50 [sort_index = sort_index.viewer().name("sort_index")] __device__(int i) mutable
51 { sort_index(i) = i; });
52
53 DeviceMergeSort().SortPairs(ij_pairs.data(),
54 sort_index.data(),
55 ij_pairs.size(),
56 [] __device__(const int2& a, const int2& b) {
57 return a.x < b.x || (a.x == b.x && a.y < b.y);
58 });
59
60 // set ij_pairs back to row_indices and col_indices
61
62 auto dst_row_indices = to.row_indices();
63 auto dst_col_indices = to.col_indices();
64
65 ParallelFor(256)
66 .kernel_name("set col row indices")
67 .apply(dst_row_indices.size(),
68 [row_indices = dst_row_indices.viewer().name("row_indices"),
69 col_indices = dst_col_indices.viewer().name("col_indices"),
70 ij_pairs = ij_pairs.viewer().name("ij_pairs")] __device__(int i) mutable
71 {
72 row_indices(i) = ij_pairs(i).x;
73 col_indices(i) = ij_pairs(i).y;
74 });
75
76 // sort the block values
77
78 loose_resize(unique_values, from.m_values.size());
79
80 ParallelFor(256)
81 .kernel_name("set block values")
82 .apply(src_values.size(),
83 [src_values = src_values.cviewer().name("blocks"),
84 sort_index = sort_index.cviewer().name("sort_index"),
85 dst_values = unique_values.viewer().name("values")] __device__(int i) mutable
86 { dst_values(i) = src_values(sort_index(i)); });
87}

122 row_indices = row_indices.viewer().name("row_indices"),

134void MatrixFormatConverter<T, 1>::make_unique_values(const DeviceTripletMatrix<T, 1>& from,
135 DeviceCOOMatrix<T>& to)
136{
137 using namespace muda;
138
139 auto& row_indices = to.m_row_indices;
140 auto& values = to.m_values;
141 values.resize(row_indices.size());
142 // first we add the offsets to counts, to get the offset_ends
143
144 ParallelFor(256)
145 .kernel_name("calculate offset_ends")
146 .apply(unique_counts.size(),
147 [offset = offsets.cviewer().name("offset"),
148 counts = unique_counts.viewer().name("counts")] __device__(int i) mutable
149 { counts(i) += offset(i); });
150
151 auto& begin_offset = offsets;
152 auto& end_offset = unique_counts; // already contains the offset_ends
153
154 // then we do a segmented reduce to get the unique blocks
155 DeviceSegmentedReduce().Sum(unique_values.data(),
156 values.data(),
157 values.size(),
158 offsets.data(),
159 end_offset.data());

205template <typename T>
206void MatrixFormatConverter<T, 1>::calculate_block_offsets(const DeviceCOOMatrix<T>& from,
207 DeviceCSRMatrix<T>& to)
208{
209 using namespace muda;
210 to.reshape(from.rows(), from.cols());
211
212 auto& dst_row_offsets = to.m_row_offsets;
213
214 // alias the offsets to the col_counts_per_row(reuse)
215 auto& col_counts_per_row = offsets;
216 col_counts_per_row.resize(to.m_row_offsets.size());
217 col_counts_per_row.fill(0);
218
219 loose_resize(unique_indices, from.non_zeros());
220 loose_resize(unique_counts, from.non_zeros());
221
222 // run length encode the row
223 DeviceRunLengthEncode().Encode(from.m_row_indices.data(),
224 unique_indices.data(),
225 unique_counts.data(),
226 count.data(),
227 from.non_zeros());
228 int h_count = count;
229
230 unique_indices.resize(h_count);
231 unique_counts.resize(h_count);
232
233 ParallelFor(256)
234 .kernel_name(__FUNCTION__)
235 .apply(unique_counts.size(),
236 [unique_indices = unique_indices.cviewer().name("offset"),
237 counts = unique_counts.viewer().name("counts"),
238 col_counts_per_row = col_counts_per_row.viewer().name(
239 "col_counts_per_row")] __device__(int i) mutable
240 {
241 auto row = unique_indices(i);
242 col_counts_per_row(row) = counts(i);
243 });
244
245 // calculate the offsets
246 DeviceScan().ExclusiveSum(col_counts_per_row.data(),
247 dst_row_offsets.data(),
