77MUDA_INLINE MUDA_DEVICE uint32_t next_idx(uint32_t* data_offset, uint32_t size, uint32_t total_size)
78{
79
80 uint32_t old = *data_offset;
81 if(old + size >= total_size)
82 return ~0u;
83 uint32_t assumed;
84 do
85 {
86 assumed = old;
87 auto new_offset = old + size;
88 old = atomic_cas(data_offset, assumed, new_offset);
89 if(old + size >= total_size)
90 {
91 old = ~0u;
92 break;
93 }
94 } while(assumed != old);
95 return old;
96}

120MUDA_INLINE MUDA_DEVICE bool LoggerViewer::push_data(details::LoggerMetaData meta,
121 const void* data)
122{
123 auto meta_idx = next_meta_data_idx();
124 if(meta_idx == ~0u)
125 {
126 MUDA_KERNEL_WARN_WITH_LOCATION(
127 "LoggerViewer: meta data is exceeded, "
128 "the content[id=%d] will be discarded.",
129 meta.id);
130 return false;
131 }
132 auto buffer_idx = next_buffer_idx(meta.size);
133 if(buffer_idx == ~0u)
134 {
135 meta.exceeded = true;
136 MUDA_KERNEL_WARN_WITH_LOCATION(
137 "LoggerViewer: log buffer is exceeded, "
138 "the content[id=%d] will be discarded.",
139 meta.id);
140
141 m_meta_data[meta_idx] = meta;
142 m_meta_data_id[meta_idx] = meta.id;
143 return false;
144 }
145 meta.offset = buffer_idx;
146 m_meta_data[meta_idx] = meta;
147 m_meta_data_id[meta_idx] = meta.id;
148 for(int i = 0; i < meta.size; ++i)
149 m_buffer[buffer_idx + i] = reinterpret_cast<const char*>(data)[i];
150 return true;
151}
