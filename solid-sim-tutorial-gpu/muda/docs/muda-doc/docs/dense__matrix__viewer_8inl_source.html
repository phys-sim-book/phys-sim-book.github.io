130template <typename T>
131template <int M, int N>
132MUDA_DEVICE Eigen::Matrix<T, M, N> DenseMatrixViewer<T>::atomic_add(const Eigen::Matrix<T, M, N>& other)
133{
134 check_size_matching(M, N);
135 Eigen::Matrix<T, M, N> ret;
136#pragma unroll
137 for(int i = 0; i < M; ++i)
138#pragma unroll
139 for(int j = 0; j < N; ++j)
140 {
141 ret(i, j) = atomic_add(i, j, other(i, j));
142 }
143 return ret;
144}
