6template <bool IsConst, typename T>
7MUDA_GENERIC auto Buffer3DViewBase<IsConst, T>::data(size_t x, size_t y, size_t z)
8 MUDA_NOEXCEPT->auto_const_t<T>*
9{
10 x += m_offset.offset_in_depth();
11 y += m_offset.offset_in_height();
12 z += m_offset.offset_in_width();
13 auto depth_begin =
14 reinterpret_cast<std::byte*>(remove_const(m_data)) + m_pitch_bytes_area * x;
15 auto height_begin = depth_begin + m_pitch_bytes * y;
16 return reinterpret_cast<T*>(height_begin) + z;

18template <bool IsConst, typename T>
19MUDA_GENERIC auto Buffer3DViewBase<IsConst, T>::data(size_t flatten_i)
20 MUDA_NOEXCEPT->auto_const_t<T>*
21{
22 auto area = m_extent.width() * m_extent.height();
23 auto x = flatten_i / area;
24 auto i_in_area = flatten_i % area;
25 auto y = i_in_area / m_extent.width();
26 auto i_in_width = i_in_area % m_extent.width();
27 auto z = i_in_width;
28 return data(x, y, z);
29}
